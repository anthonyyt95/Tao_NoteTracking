Google Sheets setup (2 minutes)

  1. Open Google Sheets, create a new spreadsheet. Copy its ID from the URL (the long string between /d/ and /edit).
  2. Open script.google.com → New project → paste this:
const SHEET_ID = 'YOUR_SPREADSHEET_ID';

function doPost(e) {
  const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet();
  const d = JSON.parse(e.postData.contents);
  // Add headers on first row if empty
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['Timestamp','Title','Category','Tags',
      'Summary','Extracted Text','Action Items','AI Provider']);
  }
  sheet.appendRow([d.scanned_at, d.title, d.category, d.tags,
    d.summary, d.extracted_text, d.action_items, d.ai_provider]);
  return ContentService
    .createTextOutput(JSON.stringify({ ok: true }))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet();
  const rows = sheet.getDataRange().getValues();
  const headers = rows.shift();
  const notes = rows.map(r => ({
    scanned_at: r[0], title: r[1], category: r[2],
    tags: r[3], summary: r[4], extracted_text: r[5],
    action_items: r[6], ai_provider: r[7]
  }));
  return ContentService
    .createTextOutput(JSON.stringify(notes))
    .setMimeType(ContentService.MimeType.JSON);
}
  1. Replace YOUR_SPREADSHEET_ID with your Sheet ID.
  2. Click Deploy → New deployment → Web app. Set Execute as: Me and Who has access: Anyone. Authorize when prompted.
  3. Copy the deployment URL → paste into the Webhook URL field above.
  4. Each time you redeploy (after edits), create a new deployment — the URL changes. Update the field.
